# 代码实现
## 用数组实现队列
```
#include<iostream>
using namespace std;

const int N = 10;
int front = -1;
int rear = -1;
int a[N];

bool Isempty()
{
    return front == -1;
}

bool Isfull()
{
    return (rear + 1) % N == front;
}

void Enqueue(int x)
{
    if (Isfull())
    {
        cout << "队列已满，无法加入" << endl;
        return;
    }
    else if (Isempty())
    {
        front = rear = 0;
    }
    else
    {
        rear = (rear + 1) % N;
    }
    a[rear] = x;
}

int Dequeue()
{
    if (Isempty())
    {
        cout << "队列已空，无法删除" << endl;
        return -1;
    }

    int value = a[front];

    if (front == rear)
    {
        // 队列中只剩一个元素，删除后重置
        front = rear = -1;
    }
    else
    {
        front = (front + 1) % N;
    }

    return value;
}

int Front()
{
    if (Isempty())
    {
        cout << "队列为空" << endl;
        return -1;
    }
    return a[front];
}

void Print()
{
    if (Isempty())
    {
       cout << "队列为空，无法打印"<<endl;
       return;
    }

    int index = front;
    while (index != rear)
    {
        cout << a[index] << " ";
        index = (index + 1) % N;
    }
    cout << a[index] << endl;
}

int main()
{
    Enqueue(2);
    Enqueue(3);
    Enqueue(4);
    Enqueue(5);
    Print();

    cout << "删除元素: " << Dequeue() << endl;
    Print();

    cout << "队首元素: " << Front() << endl;

    return 0;
}
```
## 用链表实现队列
```
#include<iostream>
using namespace std;
struct node
{
	int data;
	struct node* next;
};
struct node* front = NULL;
struct node* rear = NULL;
bool Isempty()
{
	return front == NULL;
}
void Enqueue(int x)
{
	if (Isempty())
	{
		struct node* temp = new node();
		temp->data = x;
		temp->next = NULL;
		front = rear = temp;
		return;
    }
	struct node* temp = new node();
	temp->data = x;
	temp->next = NULL;
	rear->next = temp;
	rear = temp;
	return;
}
void Dequeue()
{
	if ((Isempty()))
	{
		cout << "队列已空，无法删除" << endl;
		return;
	}
	else if (front == rear)
	{
		struct node* temp = front;
		front = rear = NULL;
		delete temp;
		return;
	}
	else
	{
		struct node* temp = front;
		front = front->next;
		delete temp;
		return;
	}
}
int Front()
{
	if (Isempty())
	{
		cout << "队列中没有元素" << endl;
		return -1;
	}
	return front->data;

}
void Print()
{
	if (Isempty())
	{
		cout << "队列为空，无法打印";
	}
	struct node* temp = front;
	while (temp != rear)
	{
		cout << temp->data << " ";
		temp = temp->next;
	}
	cout << temp->data << endl;
}
int main()
{
	Enqueue(2);
	Enqueue(3);
	Enqueue(4);
	Enqueue(5);
	Enqueue(6);
	Print();
	Dequeue();
	Dequeue();
	Print();
	return 0;
}
```
